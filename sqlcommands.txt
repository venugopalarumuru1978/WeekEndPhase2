CREATE TABLE  BIODATA(ROLLNO  INT  PRIMARY KEY,
STDNAME VARCHAR(20) NOT NULL, 
GENDER CHAR(1)  CHECK(GENDER IN ('M','F')),
BIRTHDATE  DATE, 
LOCATION VARCHAR(10) DEFAULT 'HYDERABAD');

SNAME VARCHAR(20), EXAM VARCHAR(10), 
SUB1 INT, SUB2 INT, TOTAL INT, AVGR FLOAT, RESULT CHAR(4));
